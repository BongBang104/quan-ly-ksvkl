```typescript
/**
 * Đọc scheduleData của 1 tháng, chuyển sang assignments[].
 * Format scheduleData: { "<empId>_YYYY-MM-DD": "S" | "D" | "OFF" | ... }
 */
private async _loadMacroAssignments(monthKey: string): Promise<{
  periodStart: string;
  periodEnd: string;
  assignments: any[];
}> {
  const schedule = await this.schRepo.findOne({ where: { monthKey } });
  if (!schedule) {
    return { periodStart: `${monthKey}-01`, periodEnd: `${monthKey}-01`, assignments: [] };
  }
  const data = schedule.data || {};
  const scheduleData = data.scheduleData || {};

  // Parse khoảng tháng: monthKey = "2026-06" → start="2026-06-01", end="2026-06-30"
  const [yearStr, monthStr] = monthKey.split('-');
  const year = parseInt(yearStr, 10);
  const month = parseInt(monthStr, 10);
  const lastDay = new Date(year, month, 0).getDate();
  const periodStart = `${yearStr}-${monthStr}-01`;
  const periodEnd   = `${yearStr}-${monthStr}-${String(lastDay).padStart(2, '0')}`;

  const assignments: any[] = [];
  for (const [key, shiftKind] of Object.entries(scheduleData)) {
    if (!key.includes('_')) continue;
    if (key === 'isPublished') continue;
    const sepIdx = key.lastIndexOf('_');
    const empId = key.slice(0, sepIdx);
    const date  = key.slice(sepIdx + 1);
    if (!/^\d{4}-\d{1,2}-\d{1,2}$/.test(date)) continue;
    // Chuẩn hóa YYYY-MM-DD
    const [y, m, d] = date.split('-');
    const isoDate = `${y}-${m.padStart(2, '0')}-${d.padStart(2, '0')}`;
    assignments.push({
      controller_id: empId,
      date: isoDate,
      shift_kind: String(shiftKind || 'OFF').toUpperCase(),
    });
  }

  // Bổ sung extraAssignments (nếu có)
  const extra = data.extraAssignments || {};
  for (const [key, value] of Object.entries(extra)) {
    // Pattern tương tự nếu có
  }

  return { periodStart, periodEnd, assignments };
}
```

```python
"""
fairness_macro.py
=================
Tính công bằng phân ca cấp tháng từ MacroRosterDraft (scheduleData).
Logic: đếm số ca S/D mỗi KSVKL, tính giờ làm chuẩn, giờ đêm.
"""
from __future__ import annotations

from collections import defaultdict
from datetime import date, datetime

from app.compliance.rest_compliance import RestRuleConfig, classify_shift_kind, ShiftKind
from app.routers.schemas_macro import MacroRosterDraft


DEFAULT_SHIFT_HOURS = {"S": (7, 19), "D": (19, 31)}  # 31 = 24+7 (qua nửa đêm)


def compute_fairness_from_macro(draft: MacroRosterDraft) -> dict:
    cfg = RestRuleConfig()
    per_ctrl = defaultdict(lambda: {
        "controller_id": "", "controller_name": "",
        "total_hours": 0.0, "night_hours": 0.0,
        "night_shifts": 0, "shift_count": 0, "work_days": set(),
    })

    ctrl_name_map = {str(c.id): c.name for c in (draft.controllers or [])}

    for a in draft.assignments:
        kind = (a.shift_kind or "OFF").upper()
        if kind not in DEFAULT_SHIFT_HOURS:
            continue
        hours = DEFAULT_SHIFT_HOURS[kind]
        cid = str(a.controller_id)
        ent = per_ctrl[cid]
        ent["controller_id"] = cid
        ent["controller_name"] = ctrl_name_map.get(cid, cid)
        ent["shift_count"] += 1
        ent["work_days"].add(a.date)
        duration = hours[1] - hours[0]
        ent["total_hours"] += duration

        # Tính ShiftKind để biết night
        d = date.fromisoformat(str(a.date))
        start = datetime(d.year, d.month, d.day, hours[0] % 24)
        end_h = hours[1]
        if end_h >= 24:
            from datetime import timedelta
            end = datetime(d.year, d.month, d.day, end_h - 24) + timedelta(days=1)
        else:
            end = datetime(d.year, d.month, d.day, end_h)
        sk = classify_shift_kind(start, end, cfg)
        if sk == ShiftKind.NIGHT:
            ent["night_shifts"] += 1
            ent["night_hours"] += duration

    controllers = []
    for cid, ent in per_ctrl.items():
        controllers.append({
            "controller_id": ent["controller_id"],
            "controller_name": ent["controller_name"],
            "total_hours": round(ent["total_hours"], 1),
            "night_hours": round(ent["night_hours"], 1),
            "night_shifts": ent["night_shifts"],
            "shift_count": ent["shift_count"],
            "work_days": len(ent["work_days"]),
        })

    # Tính std deviation cho tổng giờ
    if controllers:
        total_avg = sum(c["total_hours"] for c in controllers) / len(controllers)
        variance = sum((c["total_hours"] - total_avg) ** 2 for c in controllers) / len(controllers)
        std = variance ** 0.5
        max_h = max(c["total_hours"] for c in controllers)
        min_h = min(c["total_hours"] for c in controllers)
    else:
        total_avg = std = max_h = min_h = 0.0

    return {
        "controllers": sorted(controllers, key=lambda c: -c["total_hours"]),
        "avg_hours": round(total_avg, 1),
        "std_hours": round(std, 1),
        "max_hours": round(max_h, 1),
        "min_hours": round(min_h, 1),
        "max_delta_hours": round(max_h - min_h, 1),
    }
```

```jsx
  // Cả 2 trạng thái: applicant + kíp trưởng + counterparty đều thấy nút Từ chối

  return (
    <div style={{ display: 'flex', gap: 8, marginTop: 12 }}>
      {ex.status === 'pending' && isCounterparty && (
        <button onClick={handleAgree} disabled={busy}
                style={{ ...primaryBtn, backgroundColor: '#16a34a' }}>
          {busy ? 'Đang xử lý…' : '✓ Đồng ý nhận đổi ca'}
        </button>
      )}
      {ex.status === 'counterparty_agreed' && isChief && (
        <button onClick={handleApprove} disabled={busy}
                style={{ ...primaryBtn, backgroundColor: '#2563eb' }}>
          {busy ? 'Đang xử lý…' : '✓ Phê duyệt'}
        </button>
      )}
      {['pending', 'counterparty_agreed'].includes(ex.status) &&
       (isCounterparty || isChief || isApplicant) && (
        <button onClick={() => setShowRejectModal(true)} disabled={busy}
                style={{ ...primaryBtn, backgroundColor: '#dc2626' }}>
          ✗ Từ chối / Hủy
        </button>
      )}

      {showRejectModal && (
        <div style={modalOverlay}>
          <div style={modalContent}>
            <h3>Lý do từ chối / hủy yêu cầu</h3>
            <textarea
              rows={3} value={rejectReason}
              onChange={e => setRejectReason(e.target.value)}
              placeholder="Ghi rõ lý do để các bên liên quan biết..."
              style={{ width: '100%', padding: 8, border: '1px solid #cbd5e1', borderRadius: 6 }}
            />
            <div style={{ display: 'flex', gap: 8, marginTop: 12, justifyContent: 'flex-end' }}>
              <button onClick={() => setShowRejectModal(false)} disabled={busy}
                      style={{ padding: '8px 14px', backgroundColor: '#e2e8f0', border: 'none', borderRadius: 6 }}>
                Hủy
              </button>
              <button onClick={handleReject} disabled={busy || !rejectReason.trim()}
                      style={{ ...primaryBtn, backgroundColor: '#dc2626' }}>
                {busy ? 'Đang gửi…' : 'Xác nhận từ chối'}
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

const primaryBtn = {
  padding: '8px 14px', color: '#fff', border: 'none',
  borderRadius: 6, cursor: 'pointer', fontSize: 13, fontWeight: 600,
};
const modalOverlay = {
  position: 'fixed', top: 0, left: 0, right: 0, bottom: 0,
  backgroundColor: 'rgba(0,0,0,0.5)', display: 'flex',
  alignItems: 'center', justifyContent: 'center', zIndex: 1000,
};
const modalContent = {
  backgroundColor: '#fff', padding: 24, borderRadius: 8,
  width: '100%', maxWidth: 500,
};
```

Trong list render hiện tại (chỗ map qua `exchanges`), thay vì chỉ hiển thị info, thêm `<ExchangeActions ex={ex} currentUser={currentUser} onReload={loadExchanges} />`.
